---
title: Get Agent
openapi: get /test_framework/v1/aiagents/{id}/
slug: get-agent
---

## Response

<ResponseField name="id" type="integer">
  Unique identifier for the agent.
</ResponseField>

<ResponseField name="organization" type="integer">
  The organization this agent belongs to.
</ResponseField>

<ResponseField name="project" type="integer">
  The project this agent belongs to.
</ResponseField>

<ResponseField name="agent_name" type="string">
  A descriptive name for your agent.
</ResponseField>

<ResponseField name="description" type="string">
  A detailed description of what this agent does and how it should interact.
</ResponseField>

<ResponseField name="language" type="string">
  The primary language this agent uses for communication.
</ResponseField>

<ResponseField name="contact_number" type="string">
  Phone number assigned to this agent for calls.
</ResponseField>

<ResponseField name="inbound" type="boolean">
  Whether this agent handles inbound calls.
</ResponseField>
